215 
216  Count = 0;
217 
218  while ( Count <= timeout.getMaxCount() )
219  {
220  if ( this.getExists() )
221  {
222  lvbReturn = true;
223  break;
224  }
225  else
226  {
227  Thread.sleep( timeout.getPT() );
228  }
229  Count++;
230  }
231  }
232  catch(Exception e)
