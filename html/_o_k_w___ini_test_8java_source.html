503  public static String getJarExecutionDirectory()
504  {
505  String jarFile = null;
506  String jarDirectory = null;
507  int cutFileSeperator = 0;
508  int cutSemicolon = -1;
509 
510  jarFile = System.getProperty("java.class.path");
511  // Cut seperators
512  cutFileSeperator = jarFile.lastIndexOf(System.getProperty("file.separator"));
513  jarDirectory = jarFile.substring(0, cutFileSeperator);
514  // Cut semicolons
515  cutSemicolon = jarDirectory.lastIndexOf(';');
516  jarDirectory = jarDirectory.substring(cutSemicolon + 1, jarDirectory.length());
517 
518  return jarDirectory + System.getProperty("file.separator");
519  }
