88  @Override
89  public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
90  // before visiting entries in a directory we copy the directory
91  // (okay if directory already exists).
92  CopyOption[] options = (preserve) ? new CopyOption[] { COPY_ATTRIBUTES } : new CopyOption[0];
93 
94  Path newdir = target.resolve(source.relativize(dir));
95  try {
96  Files.copy(dir, newdir, options);
97  } catch (FileAlreadyExistsException x) {
98  // ignore
99  } catch (IOException x) {
100  System.err.format("Unable to create: %s: %s%n", newdir, x);
101  return SKIP_SUBTREE;
102  }
103  return CONTINUE;
104  }
105 
106  @Override
107  public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) {
108  copyFile(file, target.resolve(source.relativize(file)), prompt, preserve);
109  return CONTINUE;
110  }
111 
112  @Override
113  public FileVisitResult postVisitDirectory(Path dir, IOException exc) {
114  // fix up modification time of directory when done
115  if (exc == null && preserve) {
116  Path newdir = target.resolve(source.relativize(dir));
117  try {
118  FileTime time = Files.getLastModifiedTime(dir);
119  Files.setLastModifiedTime(newdir, time);
120  } catch (IOException x) {
121  System.err.format("Unable to copy all attributes to: %s: %s%n", newdir, x);
122  }
123  }
124  return CONTINUE;
125  }

127  @Override

144  public static void main(String[] args) throws IOException {
145  boolean recursive = false;
146  boolean prompt = false;
147  boolean preserve = false;
148 
149  // process options
150  int argi = 0;
151  while (argi < args.length) {
152  String arg = args[argi];
153  if (!arg.startsWith("-"))
154  break;
155  if (arg.length() < 2)
156  usage();
157  for (int i = 1; i < arg.length(); i++) {
158  char c = arg.charAt(i);
159  switch (c) {
160  case 'r':
161  recursive = true;
162  break;
163  case 'i':
164  prompt = true;
165  break;
166  case 'p':
167  preserve = true;
168  break;
169  default:
170  usage();
171  }
172  }
173  argi++;
174  }
175 
176  // remaining arguments are the source files(s) and the target location
177  int remaining = args.length - argi;
178  if (remaining < 2)
179  usage();
180  Path[] source = new Path[remaining - 1];
181  int i = 0;
182  while (remaining > 1) {
183  source[i++] = Paths.get(args[argi++]);
184  remaining--;
185  }
186  Path target = Paths.get(args[argi]);
187 
188  // check if target is a directory
189  boolean isDir = Files.isDirectory(target);
190 
191  // copy each source file/directory to target
192  for (i = 0; i < source.length; i++) {
193  Path dest = (isDir) ? target.resolve(source[i].getFileName()) : target;
194 
195  if (recursive) {
196  // follow links when copying files
197  EnumSet<FileVisitOption> opts = EnumSet.of(FileVisitOption.FOLLOW_LINKS);
198  TreeCopier tc = new TreeCopier(source[i], dest, prompt, preserve);
199  Files.walkFileTree(source[i], opts, Integer.MAX_VALUE, tc);
200  } else {
201  // not recursive so source must not be a directory
202  if (Files.isDirectory(source[i])) {
203  System.err.format("%s: is a directory%n", source[i]);
204  continue;
205  }
206  copyFile(source[i], dest, prompt, preserve);
207  }
208  }
209  }
